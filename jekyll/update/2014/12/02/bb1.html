372 	private void addLove(final CommonListBean pb, boolean islike, final TextView tv) {
373 		final BBGZNetParams params = new BBGZNetParams();
374 		if (!NetWorkUtil.isOnline(mContext)) {
375 			ToastAlone.show(mContext, "网络不好");
376 			return;
377 		}
378 		if (islike) {
379 			ToastAlone.show(mContext, "已收藏");
380 			return;
381 		}
382 		params.put("origin_id", pb.product_id);
383 		params.put("origin", "1");
384 
385 		BBGZRequest v1q = new BBGZRequest(Request.Method.POST,NI.My_Love_Add_love,
386 				params.getParams(), new BBGZRequest.CallBackListener() {
387 
388 					@Override
389 					public void onErrorResponse(VolleyError error) {
390 
391 					}
392 
393 					@Override
394 					public void onResponse(String response) {
395 						try {
396 							Gson gson = new Gson();
397 							JsonObject jsonObject = gson.fromJson(response,
398 									JsonObject.class);
399 							int code = jsonObject.get("code").getAsInt();
400 							if (C.GlobalConfig.SUCCESS_CODE == code) {
401 								String t = tv.getText().toString();
402 								try {
403 									int nu = Integer.parseInt(t) + 1;
404 									t = nu + "";
405 								} catch (Exception e) {
406 									e.printStackTrace();
407 								}
408 								pb.isLiked = "true";
409 								pb.likeNum = t;
410 								tv.setText(t);
411 								ToastAlone.show(mContext, "收藏成功");
412 							} else if (0 == code) {
413 								String msg = jsonObject.get("msg")
414 										.getAsString();
415 								ToastAlone.show(mContext, msg);
416 								UploadLogUtil.uploadV1BBInterfaceError(
417 										NI.My_Love_Add_love, url,
418 										params.toString(), msg);
419 							} else {
420 								ToastAlone.show(mContext, "其他问题");
421 								UploadLogUtil.uploadV1BBInterfaceError(
422 										NI.My_Love_Add_love, url,
423 										params.toString(), "其他问题");
424 							}
425 						} catch (JsonSyntaxException e) {
426 							ToastAlone.show(mContext, "数据解析错误");
427 							UploadLogUtil.uploadV1BBInterfaceError(
428 									NI.My_Love_Add_love, url,
429 									params.toString(), "数据解析异常");
430 							e.printStackTrace();
431 						} catch (Exception e) {
432 							e.printStackTrace();
433 							ToastAlone.show(mContext, "数据异常");
434 							UploadLogUtil.uploadV1BBInterfaceError(
435 									NI.My_Love_Add_love, url,
436 									params.toString(), "数据返回异常");
437 						}
438 					}
439 
440 					String url;
441 
442 					@Override
443 					public void returnUrl(String url) {
444 						this.url = url;
445 					}
446 				});
447 		Volley.newRequestQueue(mContext).add(v1q);
448 	}
